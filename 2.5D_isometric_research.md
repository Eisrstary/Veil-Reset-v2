# 2.5D等距视角在Phaser中的实现研究

## 1. 基本概念

### 1.1 2.5D等距视角特点
- **等距投影**：物体不因距离而缩小，保持相同尺寸
- **深度排序**：基于Y坐标进行深度排序（Y越大越靠前）
- **网格系统**：通常基于菱形网格或正方形网格旋转45度
- **透视效果**：通过精灵倾斜和缩放模拟3D感

### 1.2 技术实现方式
1. **正交相机**（Orthographic Camera）
2. **深度缓冲**（Depth Buffering）
3. **视差层**（Parallax Layers）
4. **精灵批处理**（Sprite Batching）

### 2.2 Phaser 4.0.0的等距相机支持
Phaser 4.0.0引入了更完善的等距相机支持。以下是推荐的实现方式：

```typescript
import { Scene, Cameras, GameObjects } from 'phaser';

class IsometricScene extends Scene {
    private isometricCamera!: Cameras.Scene2D.Camera;
    private isometricGrid!: IsometricGridSystem;
    
    create() {
        // 1. 创建等距相机
        this.isometricCamera = this.cameras.main;
        
        // 2. 配置相机为等距投影（Phaser 4.0.0支持）
        this.isometricCamera.setProjectionMatrix(
            this.createIsometricProjectionMatrix()
        );
        
        // 3. 创建网格系统
        this.isometricGrid = new IsometricGridSystem(64, 32);
        
        // 4. 设置深度排序规则
        this.children.sort((a, b) => {
            // 基于Y坐标排序，Y值越大越靠前
            const ay = (a as GameObjects.Sprite).y;
            const by = (b as GameObjects.Sprite).y;
            return ay - by;
        });
    }
    
    private createIsometricProjectionMatrix(): number[] {
        // 创建等距投影矩阵
        // 角度：30度俯视角，45度旋转
        const angleX = 30 * Math.PI / 180;
        const angleY = 45 * Math.PI / 180;
        
        return [
            Math.cos(angleY), 0, -Math.sin(angleY) * Math.sin(angleX), 0,
            0, Math.cos(angleX), 0, 0,
            Math.sin(angleY), 0, Math.cos(angleY) * Math.cos(angleX), 0,
            0, 0, 0, 1
        ];
    }
}
```

```typescript
// 等距精灵基类
class IsometricSprite extends GameObjects.Sprite {
    private baseHeight: number = 0; // 精灵的基础高度
    
    constructor(scene: Scene, x: number, y: number, texture: string) {
        super(scene, x, y, texture);
        
        // 设置默认深度
        this.setDepth(this.calculateDepth());
        
        // 监听位置变化
        this.on('changedata-position', this.updateDepth, this);
    }
    
    // 计算深度值（考虑Y坐标和基础高度）
    private calculateDepth(): number {
        return this.y + this.baseHeight;
    }
    
    // 更新深度
    updateDepth(): void {
        this.setDepth(this.calculateDepth());
    }
    
    // 设置基础高度（用于多层建筑、高地等）
    setBaseHeight(height: number): this {
        this.baseHeight = height;
        this.updateDepth();
        return this;
    }
}

// 深度管理器
class DepthManager {
    private scene: Scene;
    private sortInterval: number = 100; // 毫秒
    private sortTimer?: number;
    
    constructor(scene: Scene) {
        this.scene = scene;
        this.startAutoSorting();
    }
    
    // 启动自动深度排序
    startAutoSorting(): void {
        this.sortTimer = window.setInterval(() => {
            this.sortAllSprites();
        }, this.sortInterval);
    }
    
    // 停止自动排序
    stopAutoSorting(): void {
        if (this.sortTimer) {
            window.clearInterval(this.sortTimer);
        }
    }
    
    // 排序所有精灵
    private sortAllSprites(): void {
        const sprites = this.scene.children.getAll('sprite') as GameObjects.Sprite[];
        sprites.sort((a, b) => {
            const depthA = a.depth || a.y;
            const depthB = b.depth || b.y;
            return depthA - depthB;
        });
        
        // 按排序顺序设置深度
        sprites.forEach((sprite, index) => {
            sprite.setDepth(index);
        });
    }
}
```

### 2.4 等距网格系统设计
```typescript
// 等距网格坐标转换系统
class IsometricGridSystem {
    // 网格尺寸配置
    tileWidth: number;  // 菱形网格宽度
    tileHeight: number; // 菱形网格高度
    tileDepth: number;  // 高度方向（Z轴）单位
    
    constructor(tileWidth: number = 64, tileHeight: number = 32, tileDepth: number = 16) {
        this.tileWidth = tileWidth;
        this.tileHeight = tileHeight;
        this.tileDepth = tileDepth;
    }
    
    // 屏幕坐标转等距网格坐标
    screenToGrid(x: number, y: number): { col: number, row: number } {
        // 等距转换公式
        const col = Math.floor((x / this.tileWidth) + (y / this.tileHeight));
        const row = Math.floor((y / this.tileHeight) - (x / this.tileWidth));
        return { col, row };
    }
    
    // 网格坐标转屏幕坐标（等距投影）
    gridToScreen(col: number, row: number, z: number = 0): { x: number, y: number } {
        // 等距投影公式
        const x = (col - row) * (this.tileWidth / 2);
        const y = (col + row) * (this.tileHeight / 2) - (z * this.tileDepth);
        return { x, y };
    }
    
    // 3D网格坐标转屏幕坐标（支持高度）
    grid3DToScreen(col: number, row: number, z: number): { x: number, y: number } {
        const basePos = this.gridToScreen(col, row, 0);
        const heightOffset = z * this.tileDepth;
        return {
            x: basePos.x,
            y: basePos.y - heightOffset
        };
    }
    
    // 获取相邻网格位置
    getNeighborGrid(col: number, row: number, direction: 'north' | 'south' | 'east' | 'west'): { col: number, row: number } {
        switch (direction) {
            case 'north': return { col: col, row: row - 1 };
            case 'south': return { col: col, row: row + 1 };
            case 'east': return { col: col + 1, row: row };
            case 'west': return { col: col - 1, row: row };
            default: return { col, row };
        }
    }
    
    // 计算网格距离（曼哈顿距离）
    gridDistance(col1: number, row1: number, col2: number, row2: number): number {
        return Math.abs(col1 - col2) + Math.abs(row1 - row2);
    }
    
    // 检查网格是否在范围内
    isInBounds(col: number, row: number, width: number, height: number): boolean {
        return col >= 0 && col < width && row >= 0 && row < height;
    }
}

// 网格可视化器（调试用）
class GridVisualizer {
    private scene: Scene;
    private graphics!: GameObjects.Graphics;
    private gridSystem: IsometricGridSystem;
    
    constructor(scene: Scene, gridSystem: IsometricGridSystem) {
        this.scene = scene;
        this.gridSystem = gridSystem;
        this.createGraphics();
    }
    
    private createGraphics(): void {
        this.graphics = this.scene.add.graphics();
        this.graphics.setDepth(9999); // 确保在最上层
    }
    
    // 绘制网格
    drawGrid(width: number, height: number, color: number = 0xffffff, alpha: number = 0.3): void {
        this.graphics.clear();
        this.graphics.lineStyle(1, color, alpha);
        
        for (let col = 0; col < width; col++) {
            for (let row = 0; row < height; row++) {
                const screenPos = this.gridSystem.gridToScreen(col, row);
                
                // 绘制菱形网格
                this.graphics.beginPath();
                this.graphics.moveTo(screenPos.x, screenPos.y - this.gridSystem.tileHeight / 2);
                this.graphics.lineTo(screenPos.x + this.gridSystem.tileWidth / 2, screenPos.y);
                this.graphics.lineTo(screenPos.x, screenPos.y + this.gridSystem.tileHeight / 2);
                this.graphics.lineTo(screenPos.x - this.gridSystem.tileWidth / 2, screenPos.y);
                this.graphics.closePath();
                this.graphics.strokePath();
            }
        }
    }
    
    // 高亮特定网格
    highlightGrid(col: number, row: number, color: number = 0xff0000, alpha: number = 0.5): void {
        const screenPos = this.gridSystem.gridToScreen(col, row);
        
        this.graphics.fillStyle(color, alpha);
        this.graphics.fillRect(
            screenPos.x - this.gridSystem.tileWidth / 2,
            screenPos.y - this.gridSystem.tileHeight / 2,
            this.gridSystem.tileWidth,
            this.gridSystem.tileHeight
        );
    }
    
    // 显示网格坐标
    showGridCoordinates(width: number, height: number): void {
        for (let col = 0; col < width; col++) {
            for (let row = 0; row < height; row++) {
                const screenPos = this.gridSystem.gridToScreen(col, row);
                
                const text = this.scene.add.text(
                    screenPos.x, 
                    screenPos.y, 
                    `${col},${row}`,
                    { 
                        fontSize: '12px', 
                        color: '#ffffff',
                        backgroundColor: '#00000080'
                    }
                );
                text.setOrigin(0.5);
                text.setDepth(10000);
            }
        }
    }
}
```

## 3. 战斗系统架构设计

### 3.1 核心组件
1. **战场管理器**（BattleManager）：管理战斗状态、回合顺序
2. **干员系统**（OperatorSystem）：干员属性、技能、状态
3. **行动系统**（ActionSystem）：移动、攻击、技能使用
4. **AI系统**（AISystem）：敌人行为决策
5. **效果系统**（EffectSystem）：状态效果、伤害计算

### 3.2 数据流程
```
用户输入 → 行动验证 → 行动执行 → 效果应用 → 状态更新 → AI响应 → 回合结束
```

### 3.3 等距视角下的战斗系统
```typescript
// 等距战场管理器
class IsometricBattleManager {
    private scene: Scene;
    private gridSystem: IsometricGridSystem;
    private battleState: BattleState;
    private operators: Map<string, BattleOperator> = new Map();
    private enemies: Map<string, BattleEnemy> = new Map();
    
    constructor(scene: Scene, gridSystem: IsometricGridSystem) {
        this.scene = scene;
        this.gridSystem = gridSystem;
        this.battleState = this.createInitialState();
    }
    
    // 创建初始战斗状态
    private createInitialState(): BattleState {
        return {
            phase: 'deploy',
            currentTurn: 1,
            activeOperatorId: null,
            actionPoints: 3,
            maxActionPoints: 3,
            selectedGrid: null,
            hoveredGrid: null,
            availableActions: [],
            turnOrder: [],
            weather: 'clear',
            timeOfDay: 'day'
        };
    }
    
    // 在网格位置放置干员
    placeOperator(operatorId: string, col: number, row: number): boolean {
        const operator = this.operators.get(operatorId);
        if (!operator || !this.isGridAvailable(col, row)) {
            return false;
        }
        
        const screenPos = this.gridSystem.gridToScreen(col, row);
        operator.setPosition(screenPos.x, screenPos.y);
        operator.setGridPosition(col, row);
        
        // 更新深度
        operator.updateDepth();
        
        return true;
    }
    
    // 检查网格是否可用
    private isGridAvailable(col: number, row: number): boolean {
        // 检查是否在边界内
        if (!this.gridSystem.isInBounds(col, row, 20, 15)) {
            return false;
        }
        
        // 检查是否有其他单位
        for (const operator of this.operators.values()) {
            if (operator.gridCol === col && operator.gridRow === row) {
                return false;
            }
        }
        
        for (const enemy of this.enemies.values()) {
            if (enemy.gridCol === col && enemy.gridRow === row) {
                return false;
            }
        }
        
        return true;
    }
    
    // 获取可移动范围
    getMovementRange(operatorId: string): { col: number, row: number }[] {
        const operator = this.operators.get(operatorId);
        if (!operator) return [];
        
        const range = operator.movementRange;
        const reachableGrids: { col: number, row: number }[] = [];
        
        // 使用BFS算法计算可到达的网格
        const start = { col: operator.gridCol!, row: operator.gridRow! };
        const visited = new Set<string>();
        const queue: { col: number, row: number, distance: number }[] = [{ ...start, distance: 0 }];
        
        visited.add(`${start.col},${start.row}`);
        
        while (queue.length > 0) {
            const current = queue.shift()!;
            
            if (current.distance <= range) {
                reachableGrids.push({ col: current.col, row: current.row });
                
                // 检查四个方向
                const directions = [
                    { col: 0, row: -1 }, // 北
                    { col: 1, row: 0 },  // 东
                    { col: 0, row: 1 },  // 南
                    { col: -1, row: 0 }  // 西
                ];
                
                for (const dir of directions) {
                    const newCol = current.col + dir.col;
                    const newRow = current.row + dir.row;
                    const key = `${newCol},${newRow}`;
                    
                    if (!visited.has(key) && this.isGridAvailable(newCol, newRow)) {
                        visited.add(key);
                        queue.push({ col: newCol, row: newRow, distance: current.distance + 1 });
                    }
                }
            }
        }
        
        return reachableGrids;
    }
    
    // 获取攻击范围
    getAttackRange(operatorId: string, skillId: string): { col: number, row: number }[] {
        const operator = this.operators.get(operatorId);
        const skill = operator?.skills.get(skillId);
        if (!operator || !skill) return [];
        
        const range = skill.range;
        const attackableGrids: { col: number, row: number }[] = [];
        
        // 根据技能范围类型计算可攻击网格
        switch (skill.rangeType) {
            case 'line':
                // 直线攻击
                for (let distance = 1; distance <= range; distance++) {
                    attackableGrids.push(
                        { col: operator.gridCol! + distance, row: operator.gridRow! },
                        { col: operator.gridCol! - distance, row: operator.gridRow! },
                        { col: operator.gridCol!, row: operator.gridRow! + distance },
                        { col: operator.gridCol!, row: operator.gridRow! - distance }
                    );
                }
                break;
                
            case 'area':
                // 区域攻击（菱形）
                for (let dCol = -range; dCol <= range; dCol++) {
                    for (let dRow = -range; dRow <= range; dRow++) {
                        if (Math.abs(dCol) + Math.abs(dRow) <= range) {
                            attackableGrids.push({
                                col: operator.gridCol! + dCol,
                                row: operator.gridRow! + dRow
                            });
                        }
                    }
                }
                break;
                
            case 'single':
                // 单体攻击
                attackableGrids.push({ col: operator.gridCol!, row: operator.gridRow! });
                break;
        }
        
        // 过滤超出边界的网格
        return attackableGrids.filter(grid => 
            this.gridSystem.isInBounds(grid.col, grid.row, 20, 15)
        );
    }
    
    // 执行攻击
    executeAttack(attackerId: string, targetId: string, skillId: string): AttackResult {
        const attacker = this.operators.get(attackerId) || this.enemies.get(attackerId);
        const target = this.operators.get(targetId) || this.enemies.get(targetId);
        
        if (!attacker || !target) {
            return { success: false, message: '无效的攻击目标' };
        }
        
        // 检查攻击是否在范围内
        const attackRange = this.getAttackRange(attackerId, skillId);
        const targetGrid = { col: target.gridCol!, row: target.gridRow! };
        
        const isInRange = attackRange.some(grid => 
            grid.col === targetGrid.col && grid.row === targetGrid.row
        );
        
        if (!isInRange) {
            return { success: false, message: '目标不在攻击范围内' };
        }
        
        // 计算伤害
        const skill = attacker.skills.get(skillId);
        if (!skill) {
            return { success: false, message: '无效的技能' };
        }
        
        const damage = this.calculateDamage(attacker, target, skill);
        target.takeDamage(damage);
        
        // 消耗行动点
        this.battleState.actionPoints -= skill.actionPointCost;
        
        // 应用技能效果
        this.applySkillEffects(attacker, target, skill);
        
        // 检查目标是否被击败
        if (target.currentHealth <= 0) {
            this.handleDefeat(target);
        }
        
        return {
            success: true,
            damage,
            critical: false, // 简化版，实际应有暴击计算
            effectsApplied: skill.effects.map(e => e.type)
        };
    }
    
    // 计算伤害
    private calculateDamage(attacker: BattleUnit, target: BattleUnit, skill: Skill): number {
        let damage = skill.baseDamage;
        
        // 考虑攻击者的攻击力
        damage += attacker.attackPower;
        
        // 考虑目标的防御力
        damage -= target.defensePower * 0.5;
        
        // 考虑技能加成
        damage *= skill.damageMultiplier;
        
        // 考虑属性克制
        const typeAdvantage = this.getTypeAdvantage(skill.element, target.element);
        damage *= typeAdvantage;
        
        // 确保最小伤害
        return Math.max(1, Math.floor(damage));
    }
    
    // 获取属性克制倍率
    private getTypeAdvantage(attackElement: string, defenseElement: string): number {
        const advantages: Record<string, string[]> = {
            'fire': ['nature', 'ice'],
            'water': ['fire', 'nature'],
            'nature': ['water', 'earth'],
            'earth': ['lightning', 'fire'],
            'lightning': ['water', 'earth']
        };
        
        if (advantages[attackElement]?.includes(defenseElement)) {
            return 1.5; // 克制
        } else if (advantages[defenseElement]?.includes(attackElement)) {
            return 0.5; // 被克
        }
        
        return 1.0; // 无克制
    }
    
    // 应用技能效果
    private applySkillEffects(attacker: BattleUnit, target: BattleUnit, skill: Skill): void {
        for (const effect of skill.effects) {
            switch (effect.type) {
                case 'burn':
                    target.addStatusEffect({
                        type: 'burn',
                        duration: effect.duration || 3,
                        damagePerTurn: effect.value || 10,
                        source: attacker.id
                    });
                    break;
                    
                case 'stun':
                    target.addStatusEffect({
                        type: 'stun',
                        duration: effect.duration || 1,
                        source: attacker.id
                    });
                    break;
                    
                case 'heal':
                    target.heal(effect.value || 0);
                    break;
                    
                case 'buff':
                    target.addBuff({
                        type: effect.buffType!,
                        value: effect.value || 0,
                        duration: effect.duration || 3,
                        source: attacker.id
                    });
                    break;
            }
        }
    }
    
    // 处理单位被击败
    private handleDefeat(unit: BattleUnit): void {
        if (this.operators.has(unit.id)) {
            this.operators.delete(unit.id);
            
            // 检查是否所有干员都被击败
            if (this.operators.size === 0) {
                this.battleState.phase = 'defeat';
            }
        } else if (this.enemies.has(unit.id)) {
            this.enemies.delete(unit.id);
            
            // 检查是否所有敌人都被击败
            if (this.enemies.size === 0) {
                this.battleState.phase = 'victory';
            }
        }
        
        // 播放击败动画
        unit.playDefeatAnimation();
    }
    
    // 结束当前回合
    endTurn(): void {
        // 应用每回合效果（如燃烧伤害）
        this.applyTurnEffects();
        
        // 更新回合数
        this.battleState.currentTurn++;
        
        // 重置行动点
        this.battleState.actionPoints = this.battleState.maxActionPoints;
        
        // 切换回合（简化版）
        if (this.battleState.phase === 'player_turn') {
            this.battleState.phase = 'enemy_turn';
            this.executeEnemyTurns();
        } else {
            this.battleState.phase = 'player_turn';
        }
    }
    
    // 执行敌人回合
    private executeEnemyTurns(): void {
        for (const enemy of this.enemies.values()) {
            // 简化AI：攻击最近的干员
            const nearestOperator = this.findNearestOperator(enemy);
            if (nearestOperator) {
                const skill = enemy.getBestSkillForTarget(nearestOperator);
                if (skill) {
                    this.executeAttack(enemy.id, nearestOperator.id, skill.id);
                }
            }
        }
        
        // 敌人回合结束后回到玩家回合
        setTimeout(() => {
            this.battleState.phase = 'player_turn';
        }, 1000);
    }
    
    // 寻找最近的干员
    private findNearestOperator(enemy: BattleEnemy): BattleOperator | null {
        let nearest: BattleOperator | null = null;
        let minDistance = Infinity;
        
        for (const operator of this.operators.values()) {
            const distance = this.gridSystem.gridDistance(
                enemy.gridCol!, enemy.gridRow!,
                operator.gridCol!, operator.gridRow!
            );
            
            if (distance < minDistance) {
                minDistance = distance;
                nearest = operator;
            }
        }
        
        return nearest;
    }
}

    activeOperatorId: string | null;
    actionPoints: number;
    maxActionPoints: number;
    selectedGrid: { col: number, row: number } | null;
    hoveredGrid: { col: number, row: number } | null;
    availableActions: string[];
    turnOrder: string[];
    weather: 'clear' | 'rain' | 'fog' | 'storm';
    timeOfDay: 'day' | 'night' | 'dusk' | 'dawn';
}

// 攻击结果接口
interface AttackResult {
    success: boolean;
    damage?: number;
    critical?: boolean;
    effectsApplied?: string[];
    message?: string;
}

    defensePower: number;
    gridCol?: number;
    gridRow?: number;
    element: string;
    skills: Map<string, Skill>;
    statusEffects: StatusEffect[];
    buffs: Buff[];
    
    setPosition(x: number, y: number): void;
    setGridPosition(col: number, row: number): void;
    takeDamage(amount: number): void;
    heal(amount: number): void;
    addStatusEffect(effect: StatusEffect): void;
    addBuff(buff: Buff): void;
```

   - 获取：残响收集、区块探索、时间恢复

   - 用途：干员生存维持
   - 获取：基地生产（稻种+15%）、探索发现
   - 消耗：每日消耗

   - 用途：建筑维护、装备制造、探索消耗
   - 获取：废墟探索、交易、生产

4. **时间资源**（Time）
   - 用途：覆幕倒计时（60天）
   - 管理：每日行动消耗、时间压力机制

## 6. 技术挑战与解决方案
